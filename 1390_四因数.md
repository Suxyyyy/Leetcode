class Solution {
public:
    int numDifferentIntegers(string s) {
        string tmp="";
        s+='e';
        set<string>ss;
        for(char c : s) {
            if(!(c>='0' && c<='9')) {
                if(!tmp.empty()) {
                    ss.insert(tmp);
                    tmp.clear();
                }
            }
            else {
                if(tmp=="0") tmp.clear(); //å‰å¯¼0
                tmp+=c;
            }
        }
        return ss.size();
    }
};
