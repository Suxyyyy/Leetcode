class Solution {
public:
    int getFactorSum(int n) {
        int cnt=0;
        int sum=0;
        //å®Œå…¨å¹³æ–¹æ•°çš„å› æ•°ä¸ªæ•°ä¸€å®šä¸ºå¥‡æ•°
        int root=sqrt(n);
        if(root*root==n) return 0;
        for(int i=1;i*i<=n;++i) {
            if(n%i==0) {
                ++cnt;
                sum+=i;
                if(cnt>4) return 0;
                if((n/i)!=i) {
                    ++cnt;
                    sum+=n/i;
                }
            }
        }
        return cnt==4?sum:0;
    }
    int sumFourDivisors(vector<int>& nums) {
        int ans=0;
        for(int& x : nums) {
            ans+=getFactorSum(x);
        }
        return ans;
    }
};
