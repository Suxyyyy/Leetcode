[[655.输出二叉树]](https://leetcode.cn/problems/print-binary-tree/description/)

```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    // dfs求树高
    int height(TreeNode* root) {
        if(!root) return 0;
        return 1+max(height(root->left),height(root->right));
    }   
    vector<vector<string>> printTree(TreeNode* root) {
        int h=height(root);
        int w=(1<<h)-1; // 列数
        vector<vector<string>>res(h,vector<string>(w,""));
        function<void(TreeNode*,int,int)>dfs=[&](TreeNode* node,int row,int col) {
        if(!node) return;
        res[row][col]=to_string(node->val);
        int nextRow=row+1;
        int hh=h-1;
        if(row == h-1) return; // 到叶子直接返回
        int diffCol=1<<(hh-nextRow);
        dfs(node->left,nextRow,col-diffCol);
        dfs(node->right,nextRow,col+diffCol);
        };
        dfs(root,0,w>>1);
        return res;
    }
};
```

树高：`h = height(root)` （递归：空节点 0，非空 1 + 左右子树最大高）

矩阵宽度：`w = (1 << h) - 1` → 等价 `2^h - 1`，位运算高效无浮点误差

根节点起始坐标：`dfs(root, 0, w >> 1)` → 第 0 行、中间列（右移 = 整除 2）

子节点列偏移：`diffCol = 1 << ((h-1)-(row+1))`

左子节点：`col - diffCol`  右子节点：`col + diffCol`
